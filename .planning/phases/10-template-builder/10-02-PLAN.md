emailTemplates router:
```javascript
emailTemplates.list.useQuery()
// Returns: EmailTemplate[] with _count.campaigns, ordered by createdAt desc

emailTemplates.getById.useQuery({ id })
// Returns: EmailTemplate { id, name, subject, htmlBody, category, isStarter, createdAt }

emailTemplates.create.useMutation({ name, subject?, htmlBody, category? })
// Returns: created EmailTemplate

emailTemplates.update.useMutation({ id, name?, subject?, htmlBody?, category? })
// Returns: updated EmailTemplate

emailTemplates.delete.useMutation({ id })
// Deletes template

emailTemplates.duplicate.useMutation({ id })
// Returns: new EmailTemplate copy with "(Copy)" suffix, isStarter: false
```

emailCampaigns router:
```javascript
emailCampaigns.list.useQuery()
// Returns: EmailCampaign[] with list {id, name, _count.subscribers}, template {id, name}, _count.sends

emailCampaigns.getById.useQuery({ id })
// Returns: EmailCampaign with list and template details

emailCampaigns.create.useMutation({ name, subject, fromName?, fromEmail?, replyTo?, listId, templateId?, htmlContent? })
// Returns: created DRAFT campaign

emailCampaigns.update.useMutation({ id, name?, subject?, fromName?, fromEmail?, replyTo?, listId?, templateId?, htmlContent? })
// Updates DRAFT campaign only

emailCampaigns.delete.useMutation({ id })
// Deletes DRAFT campaign only

emailCampaigns.preview.useQuery({ id })
// Returns: { htmlContent, subject }

emailCampaigns.suggestContent.useMutation({ templateName?, htmlContent?, campaignName? })
// Returns: { subjectLines: string[], bodySuggestion: string }

emailCampaigns.schedule.useMutation({ id, scheduledFor? })
// scheduledFor=null -> send immediately (SENDING), scheduledFor=Date -> SCHEDULED
// Returns: { status, scheduledFor }

emailLists.list.useQuery()
// Returns: EmailList[] with _count.subscribers
```
