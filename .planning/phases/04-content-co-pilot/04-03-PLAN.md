               const threadParts = content.split(/\n(?=\d+[.)]\s|Tweet \d|Post \d)/);
               const newTweets = threadParts.map(t => t.replace(/^\d+[.)]\s*|^Tweet \d+[:.]\s*|^Post \d+[:.]\s*/i, '').trim()).filter(Boolean);
               if (tweets.some(t => t.trim())) {
                 // Composer has content -- ask replace or append
                 if (window.confirm('Replace current draft? (Cancel to append below)')) {
                   setTweets(newTweets.length > 0 ? newTweets : [content]);
                 } else {
                   setTweets([...tweets, ...newTweets]);

               const currentContent = tweets[0] || '';
